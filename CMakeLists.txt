cmake_minimum_required(VERSION 3.14)
project(testapp LANGUAGES CXX)

set(CMAKE_INCLUDE_CURRENT_DIR ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(Qt6 COMPONENTS Core)
find_package(Qt6 COMPONENTS Gui)
find_package(Qt6 COMPONENTS Qml)
find_package(Qt6 COMPONENTS Quick)
find_package(Qt6 COMPONENTS ShaderTools)

qt_add_executable(testapp
    main.cpp
    rhiitem.cpp rhiitem.h rhiitem_p.h
    testrhiitem.cpp testrhiitem.h
    cube.h
)

set_property(SOURCE rhiitem.cpp PROPERTY SKIP_AUTOMOC ON)

target_link_libraries(testapp PUBLIC
    Qt::Core
    Qt::Gui
    Qt::GuiPrivate
    Qt::Qml
    Qt::Quick
    Qt::QuickPrivate
)

qt_add_shaders(testapp "testapp-shaders"
    PREFIX
        "/"
    FILES
        "texture.vert"
        "texture.frag"
)

qt_add_qml_module(testapp
    URI TestApp
    VERSION 1.0
    QML_FILES main.qml
    NO_RESOURCE_TARGET_PATH
)
